    void Start()
    {
        quizPanel.SetActive(false);
        StartCoroutine(StartGPS());
    }

    IEnumerator StartGPS()
    {
#if UNITY_ANDROID || UNITY_IOS
        if (!Input.location.isEnabledByUser)
        {
            locationText.text = "GPS désactivé";
            yield break;
        }

        Input.location.Start(desiredAccuracyMeters, updateDistanceMeters);

        int maxWait = 15;
        while (Input.location.status == LocationServiceStatus.Initializing && maxWait-- > 0)
            yield return new WaitForSeconds(1);

        if (Input.location.status != LocationServiceStatus.Running)
        {
            locationText.text = "GPS indisponible";
            yield break;
        }

        locationText.text = "GPS actif";

        while (true)
        {
            var d = Input.location.lastData;

            if (d.horizontalAccuracy > maxAcceptedAccuracy)
            {
                locationText.text = $"GPS imprécis ({d.horizontalAccuracy:F1} m)";
                yield return new WaitForSeconds(refreshSeconds);
                continue;
            }

            playerLatitude = d.latitude;
            playerLongitude = d.longitude;

            DetectPlayerLocation();

            yield return new WaitForSeconds(refreshSeconds);
        }
#else
        locationText.text = "GPS non dispo sur PC";
        yield break;
#endif
    }

    void DetectPlayerLocation()
    {
        if (quizActive) return;

        float closestDistance = float.MaxValue;
        int closestIndex = -1;

        for (int i = 0; i < locations.Count; i++)
        {
            float distance = HaversineDistance(
                playerLatitude, playerLongitude,
                locations[i].latitude, locations[i].longitude
            );

            if (distance <= detectionRadiusMeters && distance < closestDistance)
            {
                closestDistance = distance;
                closestIndex = i;
            }
        }

        if (closestIndex != -1)
        {
            quizActive = true;
            currentLocationIndex = closestIndex;
            currentQuestionIndex = 0;

            locationText.text = "Lieu : " + locations[currentLocationIndex].nom;
            ShowQuestion();
        }
        else
        {
            locationText.text = "Aucun lieu détecté";
        }
    }

    void ShowQuestion()
    {
        quizPanel.SetActive(true);
        feedbackText.text = "";

        Question q = locations[currentLocationIndex].questions[currentQuestionIndex];
        questionText.text = q.texte;

        yesButton.onClick.RemoveAllListeners();
        noButton.onClick.RemoveAllListeners();

        yesButton.onClick.AddListener(() => CheckAnswer(true));
        noButton.onClick.AddListener(() => CheckAnswer(false));
    }

    void CheckAnswer(bool choix)
    {
        Question q = locations[currentLocationIndex].questions[currentQuestionIndex];

        if (choix == q.reponseOui)
        {
            currentQuestionIndex++;

            if (currentQuestionIndex >= locations[currentLocationIndex].questions.Length)
            {
                feedbackText.text = "Lieu terminé !";
                quizPanel.SetActive(false);
                quizActive = false;
            }
            else
            {
                feedbackText.text = "Bonne réponse !";
                ShowQuestion();
            }
        }
        else
        {
            feedbackText.text = "Mauvaise réponse";
        }
    }

    float HaversineDistance(float lat1, float lon1, float lat2, float lon2)
    {
        float R = 6371000f;
        float dLat = Mathf.Deg2Rad * (lat2 - lat1);
        float dLon = Mathf.Deg2Rad * (lon2 - lon1);

        float a =
            Mathf.Sin(dLat / 2) * Mathf.Sin(dLat / 2) +
            Mathf.Cos(Mathf.Deg2Rad * lat1) *
            Mathf.Cos(Mathf.Deg2Rad * lat2) *
            Mathf.Sin(dLon / 2) * Mathf.Sin(dLon / 2);

        float c = 2 * Mathf.Atan2(Mathf.Sqrt(a), Mathf.Sqrt(1 - a));
        return R * c;
    }

    class Location
    {
        public string nom;
        public float latitude;
        public float longitude;
        public Question[] questions;

        public Location(string nom, float lat, float lon, Question[] questions)
        {
            this.nom = nom;
            this.latitude = lat;
            this.longitude = lon;
            this.questions = questions;
        }
    }

    class Question
    {
        public string texte;
        public bool reponseOui;

        public Question(string texte, bool reponseOui)
        {
            this.texte = texte;
            this.reponseOui = reponseOui;
        }
    }
}
